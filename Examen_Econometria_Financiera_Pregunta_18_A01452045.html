def decimal_to_period(dec):
    """Convert format like 1970.1 to period 1970Q1"""
    year = int(dec)
    quarter = int(round((dec - year) * 10))   # .1  1, .2  2, 
    return pd.Period(year=year, quarter=quarter, freq='Q')

def unit_root_tests(series, name):
    """Perform ADF and KPSS tests with detailed interpretation"""
    print(f"\n{''*70}")
    print(f"Unit Root Tests for {name}")
    print(''*70)

    # ADF Test (H0: unit root exists = non-stationary)
    adf_result = adfuller(series, autolag='AIC')
    print("\n ADF Test (Augmented Dickey-Fuller):")
    print(f"   ADF Statistic: {adf_result[0]:.4f}")
    print(f"   p-value: {adf_result[1]:.4f}")
    print(f"   Critical values: 1%={adf_result[4]['1%']:.3f}, 5%={adf_result[4]['5%']:.3f}, 10%={adf_result[4]['10%']:.3f}")
    print("\n   Interpretation:")
    if adf_result[1] < 0.05:
        print(f"    p-value < 0.05: REJECT H0  {name} is STATIONARY")
    else:
        print(f"    p-value  0.05: FAIL TO REJECT H0  {name} is NON-STATIONARY")

    # KPSS Test (H0: series is stationary)
    kpss_result = kpss(series, regression='ct', nlags='auto')
    print("\n KPSS Test:")
    print(f"   KPSS Statistic: {kpss_result[0]:.4f}")
    print(f"   p-value: {kpss_result[1]:.4f}")
    print(f"   Critical values: 10%={kpss_result[3]['10%']:.3f}, 5%={kpss_result[3]['5%']:.3f}, 1%={kpss_result[3]['1%']:.3f}")
    print("\n   Interpretation:")
    if kpss_result[1] < 0.05:
        print(f"    p-value < 0.05: REJECT H0  {name} is NON-STATIONARY")
    else:
        print(f"    p-value  0.05: FAIL TO REJECT H0  {name} is STATIONARY")

    return adf_result[1], kpss_result[1]

def find_best_arima(series, max_p=5, max_d=2, max_q=5, seasonal=False, max_P=2, max_Q=2):
    """Find best ARIMA model using grid search"""
    best_aic = float('inf')
    best_bic = float('inf')
    best_order = None
    best_seasonal = None
    results_list = []

    print("\nSearching for best model...")

    for p in range(max_p + 1):
        for d in range(max_d + 1):
            for q in range(max_q + 1):
                if seasonal:
                    for P in range(max_P + 1):
                        for Q in range(max_Q + 1):
                            try:
                                model = SARIMAX(series, order=(p, d, q),
                                              seasonal_order=(P, d, Q, 4),
                                              enforce_stationarity=False,
                                              enforce_invertibility=False)
                                results = model.fit(disp=False)
                                results_list.append({
                                    'order': (p, d, q),
                                    'seasonal': (P, d, Q, 4),
                                    'AIC': results.aic,
                                    'BIC': results.bic
                                })
                                if results.aic < best_aic:
                                    best_aic = results.aic
                                    best_bic = results.bic
                                    best_order = (p, d, q)
                                    best_seasonal = (P, d, Q, 4)
                            except:
                                continue
                else:
                    try:
                        model = ARIMA(series, order=(p, d, q))
                        results = model.fit()
                        results_list.append({
                            'order': (p, d, q),
                            'AIC': results.aic,
                            'BIC': results.bic
                        })
                        if results.aic < best_aic:
                            best_aic = results.aic
                            best_bic = results.bic
                            best_order = (p, d, q)
                    except:
                        continue

    # Display top 10 models
    results_df = pd.DataFrame(results_list).sort_values('AIC').head(10)
    print("\nTop 10 models by AIC:")
    print(results_df.to_string(index=False))

    print(f"\n{''*70}")
    print("BEST MODEL:")
    print(f"  Order (p,d,q): {best_order}")
    if seasonal:
        print(f"  Seasonal (P,D,Q,s): {best_seasonal}")
    print(f"  AIC: {best_aic:.2f}")
    print(f"  BIC: {best_bic:.2f}")
    print(f"{''*70}")

    return best_order, best_seasonal if seasonal else None

# Find best ARIMA model

def diagnose_residuals(model, model_name):
    """Comprehensive residual diagnostics"""
    residuals = model.resid

    print(f"\n{''*70}")
    print(f"RESIDUAL DIAGNOSTICS FOR {model_name}")
    print(''*70)

    # Summary statistics
    print("\n Residual Summary Statistics:")
    print(f"   Mean: {residuals.mean():.6f} (should be  0)")
    print(f"   Std Dev: {residuals.std():.6f}")
    print(f"   Skewness: {residuals.skew():.4f} (should be  0)")
    print(f"   Kurtosis: {residuals.kurtosis():.4f} (should be  3)")

    # Ljung-Box test
    print("\n Ljung-Box Test for Autocorrelation:")
    lb_test = acorr_ljungbox(residuals, lags=[10, 15, 20], return_df=True)
    print(lb_test)
    print("\n   Interpretation:")
    print("   H0: Residuals are independently distributed (white noise)")
    print("   If p-value > 0.05: FAIL TO REJECT H0  Residuals are white noise ")
    print("   If p-value < 0.05: REJECT H0  Residuals show autocorrelation ")

    # Plot diagnostics
    fig, axes = plt.subplots(2, 2, figsize=(15, 10))

    # Residuals over time
    axes[0, 0].plot(residuals.index.to_timestamp(), residuals.values)
    axes[0, 0].axhline(y=0, color='r', linestyle='--')
    axes[0, 0].set_title(f'Residuals Over Time - {model_name}')
    axes[0, 0].set_xlabel('Time')
    axes[0, 0].set_ylabel('Residuals')

    # Histogram
    axes[0, 1].hist(residuals, bins=30, edgecolor='black', alpha=0.7)
    axes[0, 1].set_title('Histogram of Residuals')
    axes[0, 1].set_xlabel('Residuals')
    axes[0, 1].set_ylabel('Frequency')

    # ACF
    plot_acf(residuals, lags=40, ax=axes[1, 0])
    axes[1, 0].set_title('ACF of Residuals')

    # Q-Q plot
    sm.qqplot(residuals, line='s', ax=axes[1, 1])
    axes[1, 1].set_title('Q-Q Plot')

    plt.tight_layout()
    plt.show()

    # Interpretation
    print("\n Overall Assessment:")
    mean_ok = abs(residuals.mean()) < 0.01
    lb_ok = all(lb_test['lb_pvalue'] > 0.05)

    if mean_ok and lb_ok:
        print("    Residuals appear to be WHITE NOISE")
        print("    Model is adequate")
    else:
        print("    Residuals show patterns")
        print("    Model may need improvement")

def forecast_and_plot(model, series, model_name, steps=4):
    """Generate and plot forecasts with confidence intervals"""

    # Generate forecast
    forecast_obj = model.get_forecast(steps=steps)
    forecast_mean = forecast_obj.predicted_mean
    forecast_ci = forecast_obj.conf_int(alpha=0.05)  # 95% CI

    # Create forecast index
    last_period = series.index[-1]
    forecast_index = pd.period_range(start=last_period + 1, periods=steps, freq='Q')

    # Plot
    plt.figure(figsize=(14, 7))

    # Historical data
    plt.plot(series.index.to_timestamp(), series.values, label='Historical Data',
             linewidth=2, color='darkblue')

    # Forecast
    plt.plot(forecast_index.to_timestamp(), forecast_mean, label='Forecast',
             linewidth=2, color='red', linestyle='--', marker='o')

    # Confidence interval
    plt.fill_between(forecast_index.to_timestamp(),
                     forecast_ci.iloc[:, 0],
                     forecast_ci.iloc[:, 1],
                     color='pink', alpha=0.3, label='95% Confidence Interval')

    plt.title(f'{model_name} - Forecast for Next 4 Quarters', fontsize=14, fontweight='bold')
    plt.xlabel('Quarter')
    plt.ylabel('LOG PDI')
    plt.legend(loc='best')
    plt.grid(True, alpha=0.3)
    plt.tight_layout()
    plt.show()

    # Print forecast table
    forecast_df = pd.DataFrame({
        'Quarter': forecast_index,
        'Forecast': forecast_mean.values,
        'Lower 95% CI': forecast_ci.iloc[:, 0].values,
        'Upper 95% CI': forecast_ci.iloc[:, 1].values
    })

    print(f"\n{''*70}")
    print(f"{model_name} FORECAST TABLE")
    print(''*70)
    print(forecast_df.to_string(index=False))

    # Growth rate
    last_value = series.iloc[-1]
    print(f"\nLast observed value: {last_value:.4f}")
    print(f"Forecast for Q+4: {forecast_mean.iloc[-1]:.4f}")
    print(f"Change over 4 quarters: {forecast_mean.iloc[-1] - last_value:.4f}")
    print(f"Implied growth rate: {((forecast_mean.iloc[-1] - last_value) / 4 * 100):.4f}% per quarter")

    return forecast_df
